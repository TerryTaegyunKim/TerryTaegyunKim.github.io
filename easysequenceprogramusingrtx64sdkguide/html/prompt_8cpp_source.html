6std::string Prompt::ExecCommand(std::string szCmdArg)
7{
8 std::string szResult;
9
10 HANDLE hChildStdoutRd;
11 HANDLE hChildStdoutWr;
12
13 BOOL fSuccess;
14
15 // Create security attributes to create pipe.
16 SECURITY_ATTRIBUTES saAttr = { sizeof(SECURITY_ATTRIBUTES) };
17 saAttr.bInheritHandle = TRUE;
18 saAttr.lpSecurityDescriptor = NULL;
19
20 // Create a pipe to get results from child's stdout.
21 if (!CreatePipe(&hChildStdoutRd, &hChildStdoutWr, &saAttr, 0))
22 {
23 return szResult;
24 }
25
26 STARTUPINFOA si = { sizeof(STARTUPINFO) };
27
28 si.dwFlags = STARTF_USESHOWWINDOW | STARTF_USESTDHANDLES;
29 si.hStdOutput = hChildStdoutWr;
30 si.hStdError = hChildStdoutWr;
31 si.wShowWindow = SW_HIDE;
32
33 PROCESS_INFORMATION pi = { 0 };
34
35 // Create the child process.
36 fSuccess = CreateProcessA(
37 NULL,
38 (LPSTR)szCmdArg.c_str(), // command line
39 NULL, // process security attributes
40 NULL, // primary thread security attributes
41 TRUE, // TRUE=handles are inherited. Required.
42 CREATE_NEW_CONSOLE, // creation flags
43 NULL, // use parent's environment
44 NULL, // use parent's current directory
45 &si, // __in, STARTUPINFO pointer
46 &pi); // __out, receives PROCESS_INFORMATION
47
48 if (!fSuccess)
49 {
50 return szResult;
51 }
52
53 // Wait until child processes exit. Don't wait forever.
54 WaitForSingleObject(pi.hProcess, 2000);
55 TerminateProcess(pi.hProcess, 0);
56
57 if (!CloseHandle(hChildStdoutWr))
58 {
59 return szResult;
60 }
61
62 char chBuf[4096];
63 DWORD dwRead = 0;
64
65 // Read all available data from pipe
66 while (ReadFile(hChildStdoutRd, chBuf, sizeof(chBuf), &dwRead, NULL) && dwRead != 0)
67 {
68 szResult.append(chBuf, dwRead);
69 }
70
71 CloseHandle(hChildStdoutRd);
72 CloseHandle(pi.hProcess);
73 CloseHandle(pi.hThread);
74
75 return szResult;
76}
